## Install

### npm

npm install rusticflow-ui

### pnpm

pnpm add rusticflow-ui

## Usage

```jsx
// Include components

import { Box } from "rusticflow-ui"

// Add the CSS file

import "rusticflow-ui/dist/cjs/index.css"

```

```jsx
const App = () => {
  return (
    <Box>
      <div className="content">
        <div class="content">
        <h1 class="title">
          <div class="card"></div>
        </h1>
        </div>
      </div>
    </Box>
  )
}
```
